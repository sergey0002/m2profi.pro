function generate_editor2(obj)
{							
		var editor = ace.edit("fw_editor"); // теперь обращаться к редактору будем через editor
		editor.getSession().setMode("ace/mode/php");
		editor.setTheme("ace/theme/monokai");
		editor.setOptions({
		highlightSelectedWord:true, /* Подсветка выделенного слова */
		highlightActiveLine:true,
		 autoScrollEditorIntoView: true,
		copyWithEmptySelection: true,		 
		showPrintMargin : false,
		autoScrollEditorIntoView: true,
		copyWithEmptySelection: true,
		vScrollBarAlwaysVisible:true,
		});
		 
		editor.find('needle',{
		backwards: false,
		wrap: true, /*ПОиск по всему доку*/
		caseSensitive: false,
		wholeWord: false,
		regExp: false,
		preventScroll:true 
	});
	editor.session.mergeUndoDeltas = true; 			
	editor.gotoLine(1);		
}
